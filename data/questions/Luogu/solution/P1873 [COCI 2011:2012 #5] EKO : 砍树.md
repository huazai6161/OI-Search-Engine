# P7766 [COCI 2011/2012 #5] EKO 题解

看到本题，最大值最小化，很容易想到二分。

我们可以二分 $h$，每次 $O(n)$$h$ 这次计算以 $h$ 为高度的所有可以砍掉的和作为判断条件。

处理一下细节就行了。

```cpp
#include <cstdio>
#define max(a,b) ((a)>(b)?(a):(b))
typedef long long ll;
using namespace std;
const int N=1e6+7;

ll a[N];

ll l,r,mid,ans;
int n,m;

inline bool check(ll h) {
    ll res=0;
    for(int i=1;i<=n;++i)
        if(a[i]>h)
            res+=a[i]-h;
    return res>=m;
}

signed main() {
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;++i) {
        scanf("%lld",&a[i]);
        r=max(r,a[i]);
    }
    while(l<r) {
        mid=(l+r)>>1;
        if(check(mid))
            l=mid+1,ans=mid;
        else
            r=mid-1;
    }
    printf("%lld\n",ans);
    return 0;
}
```
